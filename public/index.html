1. SQL Basics

What is SQL?

SQL (Structured Query Language) is a programming language designed for managing data in relational database management systems (RDBMS).

Basic SELECT Statement

-- Select all columns from a table
SELECT * FROM employees;

-- Select specific columns
SELECT first_name, last_name, salary FROM employees;

-- Select with condition
SELECT first_name, last_name FROM employees WHERE salary > 50000;

Sample Data

emp_id first_name last_name department salary
101 John Doe IT 60000
102 Jane Smith HR 55000
103 Michael Johnson IT 65000

Practice Exercise

Write a SQL query to select the first name and last name of all employees in the IT department.

-- Solution
SELECT first_name, last_name
FROM employees
WHERE department = 'IT';

2. Filtering and Sorting

WHERE Clause with Operators

-- Comparison operators
SELECT * FROM products WHERE price > 100;

-- Logical operators (AND, OR, NOT)
SELECT * FROM employees WHERE department = 'IT' AND salary > 60000;

-- IN operator
SELECT * FROM customers WHERE country IN ('USA', 'Canada', 'Mexico');

-- LIKE operator for pattern matching
SELECT * FROM employees WHERE first_name LIKE 'J%'; -- Names starting with J
SELECT * FROM employees WHERE last_name LIKE '%son'; -- Names ending with son

ORDER BY Clause

-- Sort in ascending order (default)
SELECT * FROM employees ORDER BY salary;

-- Sort in descending order
SELECT * FROM employees ORDER BY salary DESC;

-- Sort by multiple columns
SELECT * FROM employees ORDER BY department ASC, salary DESC;

Practice Exercise

Write a SQL query to select all employees from the IT department with a salary greater than 55000, ordered by salary in descending order.

-- Solution
SELECT * FROM employees
WHERE department = 'IT' AND salary > 55000
ORDER BY salary DESC;

3. Aggregation and Grouping

Aggregate Functions

-- Count rows
SELECT COUNT(*) FROM employees;

-- Sum of values
SELECT SUM(salary) FROM employees;

-- Average value
SELECT AVG(salary) FROM employees;

-- Maximum value
SELECT MAX(salary) FROM employees;

-- Minimum value
SELECT MIN(salary) FROM employees;

GROUP BY Clause

-- Group by department and calculate average salary
SELECT department, AVG(salary) as avg_salary
FROM employees
GROUP BY department;

-- Group by multiple columns
SELECT department, job_title, AVG(salary) as avg_salary
FROM employees
GROUP BY department, job_title;

HAVING Clause

-- Filter groups based on aggregate values
SELECT department, AVG(salary) as avg_salary
FROM employees
GROUP BY department
HAVING AVG(salary) > 60000;

Practice Exercise

Write a SQL query to find departments with more than 5 employees and display the department name and employee count.

-- Solution
SELECT department, COUNT(*) as employee_count
FROM employees
GROUP BY department
HAVING COUNT(*) > 5;

4. Joins

INNER JOIN

-- Join employees with departments
SELECT e.first_name, e.last_name, d.department_name
FROM employees e
INNER JOIN departments d ON e.department_id = d.department_id;

LEFT JOIN

-- Get all employees and their department (if exists)
SELECT e.first_name, e.last_name, d.department_name
FROM employees e
LEFT JOIN departments d ON e.department_id = d.department_id;

RIGHT JOIN and FULL OUTER JOIN

-- Get all departments and their employees (if exists)
SELECT e.first_name, e.last_name, d.department_name
FROM employees e
RIGHT JOIN departments d ON e.department_id = d.department_id;

-- Get all records from both tables
SELECT e.first_name, e.last_name, d.department_name
FROM employees e
FULL OUTER JOIN departments d ON e.department_id = d.department_id;

Practice Exercise

Write a SQL query to display employee names along with their project names. Assume we have an employees table and a projects table with a common employee_id field.

-- Solution
SELECT e.first_name, e.last_name, p.project_name
FROM employees e
INNER JOIN projects p ON e.employee_id = p.employee_id;

5. Subqueries and Advanced Concepts

Subqueries in WHERE Clause

-- Find employees with salary greater than average
SELECT first_name, last_name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

Subqueries in FROM Clause

-- Use subquery as a derived table
SELECT department, avg_salary
FROM (
  SELECT department, AVG(salary) as avg_salary
  FROM employees
  GROUP BY department
) AS dept_avg
WHERE avg_salary > 60000;

Common Table Expressions (CTEs)

-- Using WITH clause for CTE
WITH high_earners AS (
  SELECT first_name, last_name, salary
  FROM employees
  WHERE salary > 70000
)
SELECT * FROM high_earners ORDER BY salary DESC;

Practice Exercise

Write a SQL query using a subquery to find employees who earn more than the average salary in their department.

-- Solution
SELECT first_name, last_name, department, salary
FROM employees e1
WHERE salary > (
  SELECT AVG(salary)
  FROM employees e2
  WHERE e2.department = e1.department
);
